cmake_minimum_required(VERSION 3.15)
project(Projeto GRPC)

set(CMAKE_MSVC_RUNTIME_LIBRARY "MultiThreaded$<$<CONFIG:Debug>:Debug>")

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

find_package(gRPC CONFIG REQUIRED)
find_package(Protobuf CONFIG REQUIRED)

add_library(${PROJECT_NAME}_lib)

target_include_directories(${PROJECT_NAME}_lib 
    PUBLIC 
        ${CMAKE_CURRENT_SOURCE_DIR}/include
        ${CMAKE_CURRENT_SOURCE_DIR}/generated
)

target_sources(${PROJECT_NAME}_lib 
    PRIVATE
        ${CMAKE_CURRENT_SOURCE_DIR}/generated/meter.pb.cc
        ${CMAKE_CURRENT_SOURCE_DIR}/generated/meter.grpc.pb.cc
        ${CMAKE_CURRENT_SOURCE_DIR}/src/meters/Meter.cpp
        ${CMAKE_CURRENT_SOURCE_DIR}/src/meters/Phase/PhasesMeter.cpp
        ${CMAKE_CURRENT_SOURCE_DIR}/src/meters/Apolo/Apolo.cpp
        ${CMAKE_CURRENT_SOURCE_DIR}/src/meters/Apolo/Apolo6031.cpp
        ${CMAKE_CURRENT_SOURCE_DIR}/src/meters/Zeus/Zeus.cpp
        ${CMAKE_CURRENT_SOURCE_DIR}/src/meters/Zeus/Zeus8021.cpp
        ${CMAKE_CURRENT_SOURCE_DIR}/src/meters/Zeus/Zeus8023.cpp
        ${CMAKE_CURRENT_SOURCE_DIR}/src/meters/Zeus/Zeus8031.cpp
        ${CMAKE_CURRENT_SOURCE_DIR}/src/meters/Cronos/Cronos.cpp
        ${CMAKE_CURRENT_SOURCE_DIR}/src/meters/Cronos/Cronos7023.cpp
        ${CMAKE_CURRENT_SOURCE_DIR}/src/meters/Cronos/Cronos7023_L.cpp
        ${CMAKE_CURRENT_SOURCE_DIR}/src/meters/Cronos/Cronos7023_2_5.cpp
        ${CMAKE_CURRENT_SOURCE_DIR}/src/meters/Cronos/Cronos6021_L.cpp
        ${CMAKE_CURRENT_SOURCE_DIR}/src/meters/Cronos/Cronos6021_A.cpp
        ${CMAKE_CURRENT_SOURCE_DIR}/src/meters/Cronos/Cronos6003.cpp
        ${CMAKE_CURRENT_SOURCE_DIR}/src/meters/Cronos/Cronos6001_A.cpp
        ${CMAKE_CURRENT_SOURCE_DIR}/src/meters/Ares/Ares.cpp
        ${CMAKE_CURRENT_SOURCE_DIR}/src/meters/Ares/Ares7021.cpp
        ${CMAKE_CURRENT_SOURCE_DIR}/src/meters/Ares/Ares8023.cpp
        ${CMAKE_CURRENT_SOURCE_DIR}/src/meters/Ares/Ares7023.cpp
        ${CMAKE_CURRENT_SOURCE_DIR}/src/meters/Ares/Ares7031.cpp
        ${CMAKE_CURRENT_SOURCE_DIR}/src/meters/Ares/Ares8023_15.cpp
        ${CMAKE_CURRENT_SOURCE_DIR}/src/meters/Ares/Ares8023_200.cpp
        ${CMAKE_CURRENT_SOURCE_DIR}/src/core/Catalog.cpp
        ${CMAKE_CURRENT_SOURCE_DIR}/src/core/CatalogControler.cpp
        ${CMAKE_CURRENT_SOURCE_DIR}/src/server/server.cpp
        ${CMAKE_CURRENT_SOURCE_DIR}/src/service/meter_service.cpp
        ${CMAKE_CURRENT_SOURCE_DIR}/src/client/client.cpp
)

target_link_libraries(${PROJECT_NAME}_lib 
    PUBLIC 
        gRPC::grpc++ 
        protobuf::libprotobuf
        proto-objects
)

add_executable(${PROJECT_NAME})

target_sources(${PROJECT_NAME} PRIVATE ${CMAKE_CURRENT_SOURCE_DIR}/src/server/main.cpp)

target_link_libraries(${PROJECT_NAME} 
    PRIVATE 
        ${PROJECT_NAME}_lib
)

add_subdirectory(tests)
